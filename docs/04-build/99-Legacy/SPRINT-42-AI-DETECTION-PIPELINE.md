   ```python
   # backend/app/services/ai_detection/metadata_detector.py
   class MetadataDetector:
       """Detect AI tools from PR metadata."""
       
       AI_MARKERS = {
           "cursor": ["cursor", "cursor.sh"],
           "copilot": ["copilot", "github copilot", "🤖"],
           "claude_code": ["claude", "anthropic", "claude code"],
           "chatgpt": ["chatgpt", "openai", "gpt-4"],
       }
       
       async def detect(self, pr_data: dict) -> DetectionResult:
           title = pr_data.get("title", "").lower()
           body = pr_data.get("body", "").lower()
           
           for tool, markers in self.AI_MARKERS.items():
               if any(m in title or m in body for m in markers):
                   return DetectionResult(
                       detected=True,
                       tool=AIToolType(tool),
                       confidence=0.8,
                       method="metadata",
                       evidence={"matched_marker": markers}
                   )
           
           return DetectionResult(detected=False, confidence=0.0)
   ```

2. **Commit Message Detector**
   ```python
   # backend/app/services/ai_detection/commit_detector.py
   class CommitDetector:
       """Detect AI tools from commit messages."""
       
       COMMIT_PATTERNS = {
           "copilot": [
               r"co-authored-by:.*copilot",
               r"generated by copilot",
               r"\[copilot\]",
           ],
           "cursor": [
               r"generated by cursor",
               r"\[cursor\]",
               r"cursor\.sh",
           ],
       }
       
       async def detect(self, commits: List[dict]) -> DetectionResult:
           for commit in commits:
               msg = commit.get("message", "").lower()
               for tool, patterns in self.COMMIT_PATTERNS.items():
                   if any(re.search(p, msg) for p in patterns):
                       return DetectionResult(
                           detected=True,
                           tool=AIToolType(tool),
                           confidence=0.9,
                           method="commit_message",
                           evidence={"commit_sha": commit["sha"]}
                       )
           return DetectionResult(detected=False, confidence=0.0)
   ```

3. **Pattern Detector (Heuristics)**
   ```python
   # backend/app/services/ai_detection/pattern_detector.py
   class PatternDetector:
       """Detect AI-generated code patterns."""
       
       AI_PATTERNS = [
           # Over-commented code
           r"(#\s*\w+.*\n){5,}",
           # Generic variable names
           r"(data|result|output|response|item)\d+",
           # Boilerplate patterns
           r"# TODO:.*implement",
       ]
       
       async def detect(self, diff: str) -> DetectionResult:
           score = 0
           evidence = []
           
           for pattern in self.AI_PATTERNS:
               matches = re.findall(pattern, diff)
               if matches:
                   score += len(matches) * 0.1
                   evidence.append({"pattern": pattern, "count": len(matches)})
           
           return DetectionResult(
               detected=score > 0.5,
               confidence=min(score, 0.7),
               method="pattern_analysis",
               evidence=evidence
           )
   ```

4. **Combined Detection Service**
   ```python
   # backend/app/services/ai_detection/service.py
   class GitHubAIDetectionService(AIDetectionService):
       def __init__(self):
           self.detectors = [
               (MetadataDetector(), 0.4),  # 40% weight
               (CommitDetector(), 0.4),    # 40% weight
               (PatternDetector(), 0.2),   # 20% weight
           ]
       
       async def detect(
           self, pr_data: dict, commits: List[dict], diff: str
       ) -> AIDetectionResult:
           results = []
           
           # Run all detectors in parallel
           tasks = [
               d.detect(pr_data if i == 0 else commits if i == 1 else diff)
               for i, (d, _) in enumerate(self.detectors)
           ]
           results = await asyncio.gather(*tasks)
           
           # Calculate weighted confidence
           total_confidence = sum(
               r.confidence * w for r, (_, w) in zip(results, self.detectors)
           )
           
           # Determine tool (highest confidence wins)
           detected_tool = max(
               [r for r in results if r.detected],
               key=lambda x: x.confidence,
               default=None
           )
           
           return AIDetectionResult(
               is_ai_generated=total_confidence > 0.5,
               confidence=total_confidence,
               detected_tool=detected_tool.tool if detected_tool else None,
               detected_model=None,
               detection_method="combined",
               evidence={
                   "metadata": results[0].evidence,
                   "commits": results[1].evidence,
                   "patterns": results[2].evidence,
               }
           )
   ```

   ```python
   # backend/app/services/validators/base.py
   from abc import ABC, abstractmethod
   from dataclasses import dataclass
   from enum import Enum
   
   class ValidatorStatus(str, Enum):
       PASSED = "passed"
       FAILED = "failed"
       SKIPPED = "skipped"
       ERROR = "error"
   
   @dataclass
   class ValidatorResult:
       validator_name: str
       status: ValidatorStatus
       message: str
       details: dict
       duration_ms: int
       blocking: bool  # If true, failure blocks merge
   
   class BaseValidator(ABC):
       name: str
       blocking: bool = True
       
       @abstractmethod
       async def validate(
           self,
           project_id: UUID,
           pr_number: str,
           files: List[str],
           diff: str
       ) -> ValidatorResult:
           pass
   ```

3. **Validation Pipeline Service**
   ```python
   # backend/app/services/validation_pipeline.py
   from typing import List
   import asyncio
   
   class ValidationPipeline:
       def __init__(self, validators: List[BaseValidator]):
           self.validators = validators
       
       async def run(
           self,
           event_id: UUID,
           project_id: UUID,
           pr_number: str,
           files: List[str],
           diff: str
       ) -> PipelineResult:
           start_time = datetime.utcnow()
           
           # Run all validators in parallel
           tasks = [
               v.validate(project_id, pr_number, files, diff)
               for v in self.validators
           ]
           results = await asyncio.gather(*tasks, return_exceptions=True)
           
           # Handle exceptions
           processed_results = []
           for v, r in zip(self.validators, results):
               if isinstance(r, Exception):
                   processed_results.append(ValidatorResult(
                       validator_name=v.name,
                       status=ValidatorStatus.ERROR,
                       message=str(r),
                       details={},
                       duration_ms=0,
                       blocking=False
                   ))
               else:
                   processed_results.append(r)
           
           # Calculate overall status
           blocking_failures = [
               r for r in processed_results
               if r.status == ValidatorStatus.FAILED and r.blocking
           ]
           
           overall_status = (
               ValidationStatus.PASSED if not blocking_failures
               else ValidationStatus.FAILED
           )
           
           duration = (datetime.utcnow() - start_time).total_seconds() * 1000
           
           return PipelineResult(
               event_id=event_id,
               status=overall_status,
               results=processed_results,
               blocking_failures=blocking_failures,
               duration_ms=int(duration)
           )
       
       async def queue(self, event_id: UUID):
           """Queue event for async processing."""
           # Use Redis queue for background processing
           await redis.rpush("validation_queue", str(event_id))
   ```

4. **Background Worker**

   ```python
   # backend/app/jobs/validation_worker.py
   async def validation_worker():
       """Process validation queue."""
       while True:
           event_id = await redis.blpop("validation_queue", timeout=60)
           if event_id:
               await process_validation(UUID(event_id))
   
   async def process_validation(event_id: UUID):
       # Fetch event
       event = await db.get(AICodeEvent, event_id)
       
       # Update status
       event.validation_started_at = datetime.utcnow()
       event.validation_status = ValidationStatus.PENDING
       await db.commit()
       
       # Fetch PR data
       pr_data = await github_service.get_pr(
           event.project_id, event.pr_number
       )
       
       # Run pipeline
       result = await pipeline.run(
           event_id=event_id,
           project_id=event.project_id,
           pr_number=event.pr_number,
           files=pr_data["files"],
           diff=pr_data["diff"]
       )
       
       # Update event
       event.validation_status = result.status
       event.validation_completed_at = datetime.utcnow()
       event.validation_duration_ms = result.duration_ms
       event.validators_run = [r.dict() for r in result.results]
       event.blocking_policies = [f.validator_name for f in result.blocking_failures]
       await db.commit()
       
       # Comment on PR
       await github_service.comment_pr(
           event.project_id,
           event.pr_number,
           format_validation_comment(result)
       )
   ```

   ```python
   class LintValidator(BaseValidator):
       name = "lint"
       blocking = True
       
       async def validate(
           self, project_id, pr_number, files, diff
       ) -> ValidatorResult:
           start = time.time()
           
           # Run linters based on file types
           python_files = [f for f in files if f.endswith(".py")]
           ts_files = [f for f in files if f.endswith((".ts", ".tsx"))]
           
           errors = []
           
           if python_files:
               # Run ruff/black
               result = await run_command(
                   f"ruff check {' '.join(python_files)}"
               )
               if result.returncode != 0:
                   errors.extend(parse_ruff_output(result.stdout))
           
           if ts_files:
               # Run eslint
               result = await run_command(
                   f"eslint {' '.join(ts_files)}"
               )
               if result.returncode != 0:
                   errors.extend(parse_eslint_output(result.stdout))
           
           duration = int((time.time() - start) * 1000)
           
           return ValidatorResult(
               validator_name=self.name,
               status=ValidatorStatus.PASSED if not errors else ValidatorStatus.FAILED,
               message=f"Found {len(errors)} lint issues" if errors else "No lint issues",
               details={"errors": errors[:20]},  # Limit to 20
               duration_ms=duration,
               blocking=self.blocking
           )
   ```

2. **Tests Validator**
   ```python
   # backend/app/services/validators/test_validator.py
   class TestValidator(BaseValidator):
       name = "tests"
       blocking = True
       
       async def validate(
           self, project_id, pr_number, files, diff
       ) -> ValidatorResult:
           start = time.time()
           
           # Determine test command based on project config
           project = await get_project(project_id)
           test_cmd = project.config.get("test_command", "pytest")
           
           # Run tests for changed files only
           test_files = self._find_related_tests(files)
           
           if not test_files:
               return ValidatorResult(
                   validator_name=self.name,
                   status=ValidatorStatus.SKIPPED,
                   message="No related tests found",
                   details={},
                   duration_ms=0,
                   blocking=False
               )
           
           result = await run_command(
               f"{test_cmd} {' '.join(test_files)} --tb=short"
           )
           
           passed = result.returncode == 0
           test_results = parse_pytest_output(result.stdout)
           
           duration = int((time.time() - start) * 1000)
           
           return ValidatorResult(
               validator_name=self.name,
               status=ValidatorStatus.PASSED if passed else ValidatorStatus.FAILED,
               message=f"{test_results['passed']}/{test_results['total']} tests passed",
               details=test_results,
               duration_ms=duration,
               blocking=self.blocking
           )
   ```

3. **Coverage Validator**
   ```python
   # backend/app/services/validators/coverage_validator.py
   class CoverageValidator(BaseValidator):
       name = "coverage"
       blocking = False  # Warning only by default
       
       async def validate(
           self, project_id, pr_number, files, diff
       ) -> ValidatorResult:
           start = time.time()
           
           # Get policy pack threshold
           policy_pack = await get_project_policy_pack(project_id)
           threshold = policy_pack.get("coverage_threshold", 80)
           
           # Run coverage
           result = await run_command(
               "pytest --cov=. --cov-report=json"
           )
           
           coverage_data = json.loads(
               Path(".coverage.json").read_text()
           )
           
           total_coverage = coverage_data["totals"]["percent_covered"]
           
           # Check changed files coverage
           changed_coverage = []
           for f in files:
               if f in coverage_data["files"]:
                   changed_coverage.append({
                       "file": f,
                       "coverage": coverage_data["files"][f]["summary"]["percent_covered"]
                   })
           
           passed = total_coverage >= threshold
           duration = int((time.time() - start) * 1000)
           
           return ValidatorResult(
               validator_name=self.name,
               status=ValidatorStatus.PASSED if passed else ValidatorStatus.FAILED,
               message=f"Coverage: {total_coverage:.1f}% (threshold: {threshold}%)",
               details={
                   "total_coverage": total_coverage,
                   "threshold": threshold,
                   "changed_files": changed_coverage
               },
               duration_ms=duration,
               blocking=policy_pack.get("coverage_blocking", False)
           )
   ```

   ```python
   # backend/app/services/github_comment_service.py
   class GitHubCommentService:
       async def post_validation_comment(
           self,
           project_id: UUID,
           pr_number: str,
           result: PipelineResult
       ):
           comment_body = self._format_comment(result)
           
           # Check for existing comment
           existing = await self._find_existing_comment(
               project_id, pr_number
           )
           
           if existing:
               await self._update_comment(existing["id"], comment_body)
           else:
               await self._create_comment(project_id, pr_number, comment_body)
       
       def _format_comment(self, result: PipelineResult) -> str:
           template = self._load_template("validation_report.md")
   ```
